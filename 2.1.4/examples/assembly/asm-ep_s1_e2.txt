.data
         w1: .byte 1, -1, 07, 0xa, 'a'
         w2: .half 1, -1, 07, 0xa, 'a'
         w3: .word 1, -1, 07, 0x12345678, 'a'
         w4: .ascii  "hello"
         w5: .asciiz "world"
         w6: .space 5

.text
main:
            # test lb, lw
           lb  $1  w1
           lw  $3  w3

            # test sb, sh, sw
           li  $1  0xF
           sb  $1  w1
           li  $2  0xF
           sw  $2  w3
