.data
   valor: .word 0

.text
   main: # readInt(&valor)
         li   $v0 5
         syscall
         sw   $v0 valor

         # valor = valor + 1
         lw   $v0 valor
         addi $v0 $v0 1
         sw   $v0 valor

         # printInt
         li   $v0 1
         lw   $a0 valor
         syscall
